def *(other)
  
  aux=Matriz.new(@col,@fil)
  i=0
  while i < @fil
    j = 0
    while j < other.col
      k = 0
      while k < @col
        aux.valor[i][j] = aux.valor[i][j] + (self.get(i,k)*other.get(k,j))
        k=k+1
      end
      j=j+1
    end
    i=i+1
  end
  aux
end

def encontrar
  i=0
  self.fil.times do |i|
    j=0
    self.col.times do |j|
      if (yield(@valor[i][j]))
        return i,j
      end
    end
    j=j+1
  end
  i=i+1
  return nil
end

def max
  max = @valor[0][0]
  i=0
    self.fil.times do |i|
      j=0
      self.col.times do |j|
          if (@valor[i][j] > max)
              max= @valor[i][j]
          end
          j=j+1
      end
    i=i+1
  end
  max
end

def to_s
  i=0
  cadena= ""
   while i<@fil
        j=0
        while j<@col
          cadena+="#{valor[i][j]} "

        end
