def *(other)
  
  aux=Matriz.new(@col,@fil)
  i=0
  while i < @fil
    j = 0
    while j < other.col
      k = 0
      while k < @col
        aux[i][j] = (aux[i][j] + @valor[i][k])*other[i][k]
        k=k+1
      end
      j=j+1
    end
    i=i+1
  end
  aux
end

def tipo
  @cnt=0
  k=0
  while k<@fil
      r=0
        while r<@col 
          if (@valor[k][r]==0)
              @cnt=@cnt+1
          end
          r=r+1
        end
      k=k+1
  end
  if (((@fil*@col)*0.6) < @cnt)  # SI el 60% de la matriz es nula
      @bool=true
      self.initialize_dispersa(@fil,@col,@valor)
  else
      @bool=false
  end
end
