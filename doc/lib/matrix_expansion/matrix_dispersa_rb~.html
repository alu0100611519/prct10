      def cv

      r=0
      h=0
      while h<@fil
        g=0
          while g<@col
              if (@valor[h][g]!=0)
                  @elementos=@elementos+1
                  @dcol[r]=g
                  @dfil[r]=h
                  @dvalor[r]=@valor[h][g]

                  r=r+1
              end
              g=g+1
          end
          h=h+1
       end
       if ((@fil*@col*0.6) < @elementos)
         puts "ERROR tiene que ser matriz dispersa"
         @dcol=nil
         @dfil=nil
         @dvalor=nil

       end

  end

    def to_s
      k=0

                          end

                          cadena+="0 "
                    end

                end

              k=k+1
          end
      puts cadena
    end

    def  *(other)
  x=Matriz.new(@col,@fil)
    i=0
    while i < @fil
      j = 0
      while j < other.col
        k = 0
        while k < @col
          x.valor[i][j] = x.valor[i][j]+ (self.get(i,k)*other.get(k,j))
